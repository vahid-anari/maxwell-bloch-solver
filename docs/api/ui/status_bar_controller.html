 30class StatusState(str, Enum):
 31    """Represent the high-level operating mode shown in the status bar.
 32
 33    Attributes:
 34        READY: Idle state shown when no long-running task is active.
 35        IMPORTING: State shown while importing data or configuration.
 36        SAVING: State shown while saving data or configuration.
 37        SOLVING: State shown while a solve operation is running.
 38        EXPORTING: State shown while exporting results.
 39        ERROR: State shown after an error condition occurs.
 40    """
 41
 42    READY = "Ready"
 43    IMPORTING = "Importing…"
 44    SAVING = "Saving…"
 45    SOLVING = "Solving…"
 46    EXPORTING = "Exporting…"
 47    ERROR = "Error"

 50def format_compact_number(value: float) -> str:
 51    """Format a numeric value for compact display.
 52
 53    The returned string uses up to two decimal places for small values and fewer
 54    decimals for larger values to keep the status-bar text compact.
 55
 56    Args:
 57        value: Numeric value to format.
 58
 59    Returns:
 60        Formatted numeric string suitable for compact display.
 61    """
 62
 63    if value < 10:
 64        return f"{value:.2f}"
 65    if value < 100:
 66        return f"{value:.1f}"
 67    return f"{value:.0f}"

 70class StatusBarController(QObject):
 71    """Control the status-bar widgets for path, timing, and memory feedback."""
 72
 73    def __init__(
 74        self,
 75        window: QMainWindow,
 76        timer_interval: int = 700,
 77    ) -> None:
 78        """Initialize the status-bar controller.
 79
 80        Args:
 81            window: Main window whose status bar is managed by this controller.
 82            timer_interval: Update interval, in milliseconds, for refreshing the
 83                elapsed-time and memory displays while timing is active.
 84        """
 85
 86        super().__init__(window)
 87        self._window = window
 88        self._sb = window.statusBar()
 89        self._process = psutil.Process(os.getpid())
 90        self._time_ns: int = 0
 91        self._last_state = StatusState.READY
 92
 93        self._current_path: str = ""
 94        self._path_modified: bool = False
 95
 96        self._timer = QTimer(self)
 97        self._timer.setInterval(timer_interval)
 98        self._timer.timeout.connect(self._update_time)
 99
100        self._status_lbl = self._make_label("Ready")
101        self._folder_path_lbl = self._make_label("Path:")
102        self._memory_lbl = self._make_label("0.00 MB", 6, Qt.AlignRight)
103        self._time_lbl = self._make_label("0.00 s", 6, Qt.AlignRight)
104
105        self._status_lbl.setFixedWidth(70)
106        self._make_layout()
107
108    def _make_label(
109        self,
110        text: str,
111        char_width: Optional[int] = None,
112        align: Qt.Alignment = Qt.AlignLeft,
113    ) -> QLabel:
114        """Create a QLabel configured for the status bar.
115
116        Args:
117            text: Initial label text.
118            char_width: Optional approximate width in characters used to size the
119                widget.
120            align: Horizontal alignment for the label text.
121
122        Returns:
123            Configured status-bar label widget.
124        """
125
126        l = QLabel(text, alignment=align | Qt.AlignVCenter)
127        if char_width is not None:
128            set_widget_width(l, char_width)
129        return l
130
131    def _make_progress(self) -> QProgressBar:
132        """Create the indeterminate progress bar for long operations.
133
134        Returns:
135            Hidden indeterminate progress bar configured for the status bar.
136        """
137
138        progress = QProgressBar(self._sb)
139        progress.setTextVisible(True)
140        progress.setFixedWidth(200)
141        progress.setRange(0, 0)
142        progress.hide()
143        progress.setFixedHeight(18)
144        return progress
145
146    def _make_layout(self) -> None:
147        """Build and attach the left and right status-bar sections."""
148
149        left_w = QWidget()
150        left_l = QHBoxLayout(left_w)
151        left_l.setContentsMargins(0, 0, 0, 0)
152        left_l.setSpacing(0)
153        left_l.addSpacing(10)
154        left_l.addWidget(self._status_lbl)
155        left_l.addSpacing(10)
156        self._add_v_line(left_l)
157        left_l.addWidget(self._folder_path_lbl)
158        left_l.addStretch(1)
159        self._sb.addWidget(left_w)
160
161        right_w = QWidget()
162        right_l = QHBoxLayout(right_w)
163        right_l.setContentsMargins(0, 0, 0, 0)
164        right_l.setSpacing(0)
165        right_l.addStretch(1)
166        self._add_v_line(right_l)
167        right_l.addWidget(QLabel("Memory:"))
168        right_l.addWidget(self._memory_lbl)
169        self._add_v_line(right_l)
170        right_l.addWidget(QLabel("Time:"))
171        right_l.addWidget(self._time_lbl)
172        right_l.addSpacing(8)
173        self._sb.addPermanentWidget(right_w)
174
175    def _add_v_line(self, layout: QLayout) -> None:
176        """Insert a padded vertical separator into a layout.
177
178        Args:
179            layout: Layout that will receive the separator widget.
180        """
181
182        layout.addSpacing(10)
183        layout.addWidget(self._vline())
184        layout.addSpacing(10)
185
186    def _vline(self) -> QFrame:
187        """Create a sunken vertical separator line.
188
189        Returns:
190            Vertical line widget for separating status-bar sections.
191        """
192
193        line = QFrame(self._sb)
194        line.setFrameShape(QFrame.VLine)
195        line.setFrameShadow(QFrame.Sunken)
196        return line
197
198    def _update_time(self) -> None:
199        """Refresh the elapsed-time label using an auto-selected time unit."""
200
201        cur_t = perf_counter_ns()
202        time_ = (cur_t - self._time_ns) / 1.0e6
203        unit = "ms"
204        if time_ >= 1000:
205            time_ = time_ / 1000.0
206            unit = "s"
207            if time_ >= 60:
208                time_ = time_ / 60.0
209                if time_ < 60:
210                    unit = "m"
211                else:
212                    time_ = time_ / 60.0
213                    unit = "h"
214        self._time_lbl.setText(f"{format_compact_number(time_)} {unit}")
215        self._update_memory()
216
217    def _update_memory(self) -> None:
218        """Refresh the memory label with the current process RSS usage."""
219
220        value_byte = self._process.memory_info().rss
221        unit = "B"
222        if value_byte >= 1000:
223            value_byte = value_byte / 1024
224            if value_byte < 1000:
225                unit = "KB"
226            else:
227                value_byte = value_byte / 1024
228                if value_byte < 1000:
229                    unit = "MB"
230                else:
231                    value_byte = value_byte / 1024
232                    unit = "GB"
233        self._memory_lbl.setText(f"{format_compact_number(value_byte)} {unit}")
234
235    def _update_path_label(self) -> None:
236        """Refresh the displayed path text and its tooltip."""
237
238        path_text = self._current_path if self._current_path else "Untitled"
239        mark = " *" if self._path_modified else ""
240        text = f"Path{mark}: {path_text}"
241        self._folder_path_lbl.setText(text)
242        self._folder_path_lbl.setToolTip(text)
243
244    def set_state(self, state: StatusState) -> None:
245        """Set the current status-bar state.
246
247        This method updates the state label, starts or stops the elapsed-time
248        timer as needed, refreshes the memory display, and processes pending UI
249        events.
250
251        Args:
252            state: New status value to display.
253        """
254
255        self._timer.stop()
256        if self._last_state == StatusState.SOLVING:
257            self._update_time()
258        self._last_state = state
259        if state == StatusState.SOLVING:
260            self._time_ns = perf_counter_ns()
261            self._timer.start()
262
263        self._update_memory()
264        self._status_lbl.setText(state)
265        QApplication.processEvents()
266
267    def set_path(self, path: str) -> None:
268        """Update the displayed path and clear the modified marker.
269
270        Args:
271            path: File-system path to show in the status bar. An empty string is
272                displayed as ``Untitled``.
273        """
274
275        self._current_path = path
276        if path:
277            self._path_modified = False
278        self._update_path_label()
279
280    def set_path_modified(self) -> None:
281        """Mark the current path as modified and refresh the label."""
282
283        self._path_modified = True
284        self._update_path_label()
285
286    def is_modified(self) -> bool:
287        """Return whether the current path is marked as modified.
288
289        Returns:
290            ``True`` if unsaved changes are currently indicated, otherwise
291            ``False``.
292        """
293
294        return self._path_modified

297def _demo_main() -> int:
298    """Run this module as a standalone demo application.
299
300    Returns:
301        Qt application exit code.
302    """
303
304    import sys
305    from settings.app_style import set_app_style
306
307    class MainWindow(QMainWindow):
308        """Demonstration main window for the status-bar controller."""
309
310        def __init__(self) -> None:
311            """Initialize the demo window and its controls."""
312
313            super().__init__()
314            self.setWindowTitle("StatusBarController demo")
315            self.status_ctrl = StatusBarController(self)
316            self.edit_folder_path = QLineEdit(self)
317            self.status_ctrl.set_path("C:/Vahid/Download/abcdefghijklmnopqrstuvwxyz")
318            self.combo = QComboBox(self)
319            for state in StatusState:
320                self.combo.addItem(state.value, state)
321            self.combo.setFixedWidth(self.combo.sizeHint().width())
322            cb = QCheckBox("Set Modified")
323            cb.checkStateChanged.connect(self.on_modified_changed)
324            self.cb = cb
325
326            self.edit_folder_path.editingFinished.connect(self.on_folder_path_change)
327            self.combo.currentIndexChanged.connect(self.on_combo_changed)
328
329            self._make_layout()
330
331        def _make_layout(self) -> None:
332            """Create the central layout used by the demo window."""
333
334            central = QWidget(self)
335            layout = QVBoxLayout(central)
336            self.setCentralWidget(central)
337            form = QFormLayout()
338            form.addRow("Folder", self.edit_folder_path)
339            layout.addLayout(form)
340            layout.addWidget(self.combo)
341            layout.addWidget(self.cb)
342            layout.addStretch(1)
343
344        def on_folder_path_change(self) -> None:
345            """Handle updates to the folder-path editor."""
346
347            self.status_ctrl.set_path(self.edit_folder_path.text())
348
349        def on_combo_changed(self) -> None:
350            """Handle changes to the selected status value."""
351
352            self.status_ctrl.set_state(self.combo.currentData())
353
354        def on_modified_changed(self) -> None:
355            """Handle toggling of the modified-state checkbox."""
356
357            self.status_ctrl.set_path_modified()
358
359    app = QApplication(sys.argv)
360    set_app_style(app)
361
362    w = MainWindow()
363    w.resize(900, 300)
364    w.show()
365    return app.exec()
