 18class SciNumberValidator(QValidator):
 19    """Validator for integer and floating-point numeric input.
 20
 21    The validator is permissive while the user is typing. Incomplete but still
 22    plausible input is reported as intermediate instead of invalid so editing
 23    is not blocked prematurely. For floating-point values, scientific notation
 24    is supported and non-finite values such as ``inf`` and ``nan`` are rejected.
 25
 26    Attributes:
 27        _value_is_int: Whether the validator accepts integer values only.
 28        _text_to_format_value: Callable used to parse and normalize text input.
 29        _min_limit: Minimum allowed value after normalization.
 30        _max_limit: Maximum allowed value after normalization.
 31        _min_limit_inclusive: Whether the minimum bound is inclusive.
 32        _max_limit_inclusive: Whether the maximum bound is inclusive.
 33        _rx: Regular expression used for preliminary syntax validation.
 34    """
 35
 36    def __init__(
 37            self,
 38            value_is_int: bool,
 39            text_to_format_value: Callable[[str], float],
 40            min_limit: int | float | None = None,
 41            max_limit: int | float | None = None,
 42            min_limit_inclusive: bool = True,
 43            max_limit_inclusive: bool = True,
 44            parent=None,
 45    ):
 46        """Initialize the numeric validator.
 47
 48        Args:
 49            value_is_int: If ``True``, accept integer-style input only. If
 50                ``False``, accept floating-point input including scientific
 51                notation.
 52            text_to_format_value: Callable that parses text and returns the
 53                normalized numeric value used for range checking.
 54            min_limit: Optional lower bound for valid values.
 55            max_limit: Optional upper bound for valid values.
 56            min_limit_inclusive: Whether the lower bound is inclusive.
 57            max_limit_inclusive: Whether the upper bound is inclusive.
 58            parent: Optional Qt parent object.
 59
 60        Raises:
 61            ValueError: If both bounds are provided and ``min_limit`` is
 62                greater than ``max_limit``.
 63        """
 64        super().__init__(parent)
 65
 66        if min_limit is not None and max_limit is not None and min_limit > max_limit:
 67            raise ValueError("min_limit must be <= max_limit")
 68
 69        self._value_is_int = value_is_int
 70        self._text_to_format_value = text_to_format_value
 71        self._min_limit = min_limit
 72        self._max_limit = max_limit
 73        self._min_limit_inclusive = min_limit_inclusive
 74        self._max_limit_inclusive = max_limit_inclusive
 75
 76        if value_is_int:
 77            self._rx = QRegularExpression(r"^[+\-]?\d*$")
 78        else:
 79            self._rx = QRegularExpression(r"^[+\-]?((\d+(\.\d*)?)|(\.\d*)|(\d*))([eE][+\-]?\d*)?$")
 80
 81    def validate(self, s: str, pos: int):
 82        """Validate the current text during editing.
 83
 84        Empty strings and incomplete-but-plausible numeric input are treated as
 85        intermediate states so users can continue typing without immediate
 86        rejection.
 87
 88        Args:
 89            s: Current input string being validated.
 90            pos: Cursor position supplied by Qt.
 91
 92        Returns:
 93            tuple[QValidator.State, str, int]: Qt validator state together with
 94            the unchanged input string and cursor position.
 95        """
 96        t = s.strip()
 97
 98        if t == "":
 99            return (QValidator.Intermediate, s, pos)
100
101        if not self._rx.match(t).hasMatch():
102            return (QValidator.Invalid, s, pos)
103
104        try:
105            v = self._text_to_format_value(t)
106        except ValueError:
107            return (QValidator.Intermediate, s, pos)
108
109        if not self._value_is_int and not math.isfinite(v):
110            return (QValidator.Invalid, s, pos)
111
112        # Range -> Intermediate (do not block typing)
113        if self._min_limit is not None:
114            if self._min_limit_inclusive:
115                if v < self._min_limit:
116                    return (QValidator.Intermediate, s, pos)
117            else:
118                if v <= self._min_limit:
119                    return (QValidator.Intermediate, s, pos)
120
121        if self._max_limit is not None:
122            if self._max_limit_inclusive:
123                if v > self._max_limit:
124                    return (QValidator.Intermediate, s, pos)
125            else:
126                if v >= self._max_limit:
127                    return (QValidator.Intermediate, s, pos)
128
129        return (QValidator.Acceptable, s, pos)

132class NumericLineEdit(QLineEdit):
133    """Line edit for validated numeric entry with commit gating.
134
135    This widget combines a permissive validator for in-progress typing with
136    stricter checks when the user attempts to commit or leave the field. It
137    stores the last committed value, highlights invalid state, and prevents
138    focus changes until invalid input is resolved.
139
140    Signals:
141        valueChanged: Emitted with the new committed numeric value.
142
143    Attributes:
144        _value_is_int: Whether the widget stores integer values only.
145        _val_fmt: Format string used to display committed values.
146        _parse: Conversion callable used to parse values.
147        _min_limit: Optional normalized minimum bound.
148        _max_limit: Optional normalized maximum bound.
149        _min_limit_inclusive: Whether the lower bound is inclusive.
150        _max_limit_inclusive: Whether the upper bound is inclusive.
151        _filter_installed: Whether the global event filter is currently active.
152        _invalid_property: Cached invalid-state flag mirrored to a Qt property.
153        _in_refocus: Whether a refocus cycle is currently in progress.
154        _skip_next_focus_out_check: Whether the next focus-out validation should
155            be skipped.
156        _reject_pending: Internal flag reserved for rejection bookkeeping.
157        _last_value: Last successfully committed numeric value.
158    """
159
160    valueChanged = Signal(object)
161
162    def __init__(
163            self,
164            init_val: Num,
165            value_is_int: bool = False,
166            val_fmt: str = "{:0.6g}",
167            min_limit: Num | None = None,
168            max_limit: Num | None = None,
169            min_limit_inclusive: bool = True,
170            max_limit_inclusive: bool = True,
171            place_holder_text: str = "Enter a float",
172            max_length: Optional[int] = None,
173            width_chars: Optional[int] = None,
174            parent: Optional[QWidget] = None
175    ):
176        """Initialize the numeric line edit.
177
178        Args:
179            init_val: Initial committed value shown in the widget.
180            value_is_int: If ``True``, restrict input to integer values.
181            val_fmt: Format string used to render committed values.
182            min_limit: Optional lower bound for accepted values.
183            max_limit: Optional upper bound for accepted values.
184            min_limit_inclusive: Whether the lower bound is inclusive.
185            max_limit_inclusive: Whether the upper bound is inclusive.
186            place_holder_text: Placeholder text shown when the field is empty.
187            max_length: Optional maximum character count for the line edit.
188            width_chars: Optional fixed width expressed as a number of ``0``
189                characters in the current font.
190            parent: Optional parent widget.
191        """
192        super().__init__(parent)
193
194        self._value_is_int = value_is_int
195        self._val_fmt = self._normalize_val_fmt(val_fmt)
196        self._parse = int if value_is_int else float
197
198        # Store bounds
199        self._min_limit = min_limit if min_limit is None else self._value_to_formated_value(min_limit)
200        self._max_limit = max_limit if max_limit is None else self._value_to_formated_value(max_limit)
201        self._min_limit_inclusive = min_limit_inclusive
202        self._max_limit_inclusive = max_limit_inclusive
203
204        # StatusState
205        self._filter_installed = False
206        self._invalid_property = False
207        self._in_refocus = False
208        self._skip_next_focus_out_check = False
209        self._reject_pending = False
210        self._last_value: Num = init_val
211
212        self.setValidator(self._make_validator())
213        self.setAlignment(Qt.AlignRight | Qt.AlignVCenter)
214        self.installEventFilter(self)
215        self.textEdited.connect(self._on_text_edited)
216        self.editingFinished.connect(self._on_editing_finished)
217        self.setPlaceholderText(place_holder_text)
218        if max_length is not None:
219            self.setMaxLength(max_length)
220        if width_chars is not None:
221            self.set_width_chars(width_chars)
222
223        # Initialize
224        self.set_value(init_val)

228        """Create and configure the validator used by the widget.

244    def _text_to_value(self, text: str) -> Num:
245        """Parse raw text into a numeric value.
246
247        Args:
248            text: Input text to parse.
249
250        Returns:
251            Num: Parsed numeric value.
252        """
253        return self._parse(text.strip())

255    def _value_to_text(self, value: Num) -> str:
256        """Format a numeric value for display.
257
258        Args:
259            value: Numeric value to format.
260
261        Returns:
262            str: Formatted text representation.
263        """
264        return str(self._val_fmt.format(self._parse(value)))

266    def _value_to_formated_value(self, value: Num) -> Num:
267        """Normalize a numeric value through the display format.
268
269        This applies the current formatting and then parses the formatted result
270        back into a numeric value.
271
272        Args:
273            value: Numeric value to normalize.
274
275        Returns:
276            Num: Value after formatting and reparsing.
277        """
278        return self._text_to_value(self._value_to_text(value))
279
280    def _text_to_formated_value(self, text: str) -> Num:
281        """Normalize text input through parsing and formatting.
282
283        Args:
284            text: Input text to normalize.
285
286        Returns:
287            Num: Parsed value after applying the current formatting rules.
288        """
289        v = self._text_to_value(text)
290        return self._text_to_value(self._value_to_text(v))
291
292    def _normalize_val_fmt(self, val_fmt: str) -> str:
293        """Normalize the format string used for display.
294
295        Integer widgets always use ``"{:d}"``. Floating-point widgets fall back
296        to ``"{:0.6g}"`` when the provided format string cannot be applied or
297        does not produce a parseable numeric result.
298
299        Args:
300            val_fmt: Candidate format string.
301
302        Returns:
303            str: Sanitized format string suitable for display.
304        """
305        if self._value_is_int:
306            return "{:d}"
307        try:
308            text = val_fmt.format(12345.12345)
309        except Exception:
310            return "{:0.6g}"
311
312        try:
313            float(text.replace(",", "").strip())
314        except ValueError:
315            return "{:0.6g}"
316
317        return val_fmt.replace(",", "")
318
319    def _validate_value(self, value: Num) -> Num:
320        """Validate and normalize a programmatically supplied value.
321
322        Args:
323            value: Value to validate.
324
325        Returns:
326            Num: Normalized validated value.
327
328        Raises:
329            ValueError: If the value type is invalid or the normalized value is
330                outside the configured bounds.
331        """
332        if isinstance(value, bool) or not isinstance(value, (int, float)):
333            raise ValueError(f"Invalid value: {value!r}")
334
335        validated_value = self._value_to_formated_value(value)
336        if not self._in_range(validated_value):
337            raise ValueError(self._invalid_input_text_error("New value"))
338        return validated_value
339
340    def _apply_value(self, value: Num) -> None:
341        """Apply a validated value to widget state and display.
342
343        Args:
344            value: Previously validated numeric value to store and display.
345        """
346        self._last_value = value
347        text = self._value_to_text(value)
348        self._set_invalid_property(False)
349        self._set_text_safely(text)
350
351    # ----- range -----
352    def _in_range(self, v: Num) -> bool:
353        """Check whether a value satisfies the configured bounds.
354
355        Args:
356            v: Value to test.
357
358        Returns:
359            bool: ``True`` if the value is inside the allowed interval.
360        """
361        if self._min_limit is not None:
362            if v < self._min_limit:
363                return False
364            elif v == self._min_limit:
365                return self._min_limit_inclusive
366
367        if self._max_limit is not None:
368            if v > self._max_limit:
369                return False
370            elif v == self._max_limit:
371                return self._max_limit_inclusive
372
373        return True
374
375    def _invalid_input_text_error(self, value_text: str) -> str:
376        """Build a concise error message describing valid bounds.
377
378        Args:
379            value_text: Label used to describe the invalid value in the message.
380
381        Returns:
382            str: Human-readable validation error string.
383        """
384        if self._min_limit is None and self._max_limit is None:
385            return f"{value_text} is invalid."
386
387        min_text = "-inf" if self._min_limit is None else pretty_sci_text(self._min_limit, sig_digits=3)
388        max_text = "+inf" if self._max_limit is None else pretty_sci_text(self._max_limit, sig_digits=3)
389
390        left = "[" if self._min_limit_inclusive else "("
391        right = "]" if self._max_limit_inclusive else ")"
392        return f"{value_text} must be in {left}{min_text}, {max_text}{right}"
393
394    def _invalid_input_msg_error(self) -> str:
395        """Build a detailed message for the invalid-input warning dialog.
396
397        Returns:
398            str: Multi-line validation message appropriate for user display.
399        """
400        text = self.text().strip()
401
402        if text == "":
403            return "Please enter a value."
404
405        try:
406            formated_v = self._text_to_formated_value(text)
407        except ValueError:
408            return (f"Input text: '{text}'\n"
409                    f"Please enter a valid number.")
410
411        if not self._in_range(formated_v):
412            return (f"Input: '{text}'\n"
413                    f"Format: {self._val_fmt}\n"
414                    f"Value after apply format: {formated_v}\n"
415                    f"{self._invalid_input_text_error('Input value')}")
416
417        return "Please enter a valid value."

420    def _validation_state(self, text: str) -> QValidator.StatusState:
421        """Return the validator state for the provided text.
422
423        Args:
424            text: Text to validate.
425
426        Returns:
427            QValidator.State: Validation state reported by the current validator.
428        """
429        v = self.validator()
430        if v is None:
431            return QValidator.Acceptable
432        state, _, _ = v.validate(text, 0)
433        return state

436        """Return whether the current widget text can be committed.

455    def _set_text_safely(self, text: str) -> None:
456        """Set widget text while temporarily blocking signals.
457
458        Args:
459            text: Text to assign to the line edit.
460        """
461        was_blocked = self.signalsBlocked()
462        self.blockSignals(True)
463        try:
464            self.setText(text)
465        finally:
466            self.blockSignals(was_blocked)

468    # ----- slots -----
469    def _on_text_edited(self, text: str) -> None:
470        """Update invalid styling while the user edits the text.
471
472        Args:
473            text: Current edited text.
474        """
475        self._set_invalid_property(self._validation_state(text) != QValidator.Acceptable)
476
477    def _on_editing_finished(self) -> None:
478        """Commit the edited value when the current text is acceptable."""
479        if not self._can_commit():
480            return
481
482        new_value = self._text_to_formated_value(self.text())
483        if new_value != self._last_value:
484            self._last_value = new_value
485            self._set_invalid_property(False)
486            self._set_text_safely(self._value_to_text(new_value))
487            self.valueChanged.emit(new_value)
488
489    # ----- refocus / rejection -----
490    def _refocus(self) -> None:
491        """Restore focus to the widget and select all text."""
492        self._in_refocus = True
493        try:
494            self.setFocus(Qt.OtherFocusReason)
495            self.selectAll()
496        finally:
497            self._in_refocus = False

499    def _reject_exit_ui(self) -> None:
500        """Reject an invalid exit attempt and show a warning dialog."""
501        self._set_invalid_property(True)
502        show_warning("Invalid input", self._invalid_input_msg_error(), parent=self)
503        self.setText(self._value_to_text(self._last_value))
504        self._set_invalid_property(False)
505        QTimer.singleShot(0, self._refocus)

548    # ----- focus/exit gating -----
549    def focusInEvent(self, event) -> None:
550        """Handle the Qt focus-in callback.
551
552        Installs a global event filter while the widget has focus so outside
553        mouse clicks can be intercepted and treated as commit or reject events.
554
555        Args:
556            event: Qt focus event.
557        """
558        super().focusInEvent(event)
559        if not self._filter_installed:
560            QApplication.instance().installEventFilter(self)
561            self._filter_installed = True
562
563    def focusOutEvent(self, event) -> None:
564        """Handle the Qt focus-out callback.
565
566        Invalid text prevents focus loss. Depending on the focus-change reason,
567        the event may be allowed, ignored, or converted into a rejection flow.
568
569        Args:
570            event: Qt focus event.
571        """
572        if self._skip_next_focus_out_check:
573            self._skip_next_focus_out_check = False
574            super().focusOutEvent(event)
575            if self._filter_installed:
576                QApplication.instance().removeEventFilter(self)
577                self._filter_installed = False
578            return
579
580        if self._in_refocus:
581            super().focusOutEvent(event)
582            return
583
584        if event.reason() in (Qt.PopupFocusReason, Qt.ActiveWindowFocusReason, Qt.MenuBarFocusReason):
585            super().focusOutEvent(event)
586            if self._filter_installed:
587                QApplication.instance().removeEventFilter(self)
588                self._filter_installed = False
589            return
590
591        if not self._can_commit():
592            event.ignore()
593            self._reject_exit_ui()
594            return
595
596        super().focusOutEvent(event)
597        if self._filter_installed:
598            QApplication.instance().removeEventFilter(self)
599            self._filter_installed = False
600
601    def keyPressEvent(self, event) -> None:
602        """Handle the Qt key-press callback for commit-sensitive keys.
603
604        Enter, Return, Tab, and Backtab are intercepted so invalid text cannot
605        be committed or escaped accidentally.
606
607        Args:
608            event: Qt key event.
609        """
610        key = event.key()
611
612        if key in (Qt.Key_Return, Qt.Key_Enter):
613            if not self._can_commit():
614                event.accept()
615                self._reject_exit_ui()
616                return
617            super().keyPressEvent(event)
618            return
619
620        if key in (Qt.Key_Tab, Qt.Key_Backtab):
621            if not self._can_commit():
622                event.accept()
623                self._reject_exit_ui()
624                return
625            self.focusNextPrevChild(key == Qt.Key_Tab)
626            return
627
628        super().keyPressEvent(event)
629
630    def eventFilter(self, obj, event) -> bool:
631        """Filter application events while the widget has focus.
632
633        Outside mouse clicks are interpreted as exit attempts. Invalid input is
634        rejected and the click is consumed; valid input is allowed to continue.
635
636        Args:
637            obj: Watched Qt object.
638            event: Incoming Qt event.
639
640        Returns:
641            bool: ``True`` if the event is consumed by this widget, otherwise
642            ``False``.
643        """
644        if not self.hasFocus() or self._in_refocus:
645            return False
646
647        if event.type() == QEvent.MouseButtonPress and isinstance(event, QMouseEvent):
648            gp = event.globalPosition().toPoint()
649            w = QApplication.widgetAt(gp)
650            if w is None:
651                return False
652
653            if w is self or self.isAncestorOf(w):
654                return False
655
656            # Click outside: treat as exit attempt
657            if not self._can_commit():
658                self._reject_exit_ui()
659                return True  # consume click
660
661            self._skip_next_focus_out_check = True
662            self.clearFocus()
663            return False
664
665        return False
