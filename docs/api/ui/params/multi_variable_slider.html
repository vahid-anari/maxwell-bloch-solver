 27class MultiVariableSlider(FloatSlider):
 28    """Float slider that exposes one element of an internal value array at a time.
 29
 30    The widget stores per-index current values, default values, and optional
 31    special values such as ``+inf`` and ``-inf``. The selected index is shown on
 32    the base ``FloatSlider`` interface, and updates to the active slider value
 33    are written back to the corresponding array element.
 34
 35    Attributes:
 36        arrayChanged: Emitted when the stored current-value array changes. The
 37            emitted object is a copy of the internal NumPy array.
 38    """
 39
 40    arrayChanged = Signal(object)

 42    def __init__(
 43        self,
 44        label: str,
 45        html_label: str,
 46        min_val: float,
 47        max_val: float,
 48        init_vals: Sequence[float | str],
 49        init_idx: int = 0,
 50        unit: str = "",
 51        use_same_values: bool = False,
 52        steps: int = 10000,
 53        val_fmt: str = "{:.3f}",
 54        min_limit: float | None = None,
 55        max_limit: float | None = None,
 56        min_limit_inclusive: bool = True,
 57        max_limit_inclusive: bool = True,
 58        show_range: bool = True,
 59        orientation: Qt.Orientation = Qt.Horizontal,
 60        editable: bool = True,
 61        parent: Optional[QWidget] = None,
 62    ):
 63        """Initialize the multi-variable slider.
 64
 65        Args:
 66            label: LaTeX-style label shown beside the slider.
 67            html_label: HTML label used in the values dialog table header.
 68            min_val: Minimum editable numeric range.
 69            max_val: Maximum editable numeric range.
 70            init_vals: Initial per-index values. Entries may be finite numbers or
 71                special infinity markers.
 72            init_idx: Initial active index in the stored array.
 73            unit: Optional unit label shown beside the value.
 74            use_same_values: If ``True``, changing one entry propagates the same
 75                value to all entries.
 76            steps: Integer resolution of the underlying float slider.
 77            val_fmt: Display format string for finite values.
 78            min_limit: Optional lower hard limit.
 79            max_limit: Optional upper hard limit.
 80            min_limit_inclusive: Whether the lower hard limit is inclusive.
 81            max_limit_inclusive: Whether the upper hard limit is inclusive.
 82            show_range: Whether min/max range labels are shown.
 83            orientation: Slider orientation.
 84            editable: Whether right-click editing actions are enabled.
 85            parent: Optional parent widget.
 86
 87        Raises:
 88            ValueError: If ``init_vals`` is empty, contains a finite value outside
 89                the editable range, or contains an infinity marker that conflicts
 90                with a corresponding hard limit.
 91        """
 92        init_vals = list(init_vals)
 93        if len(init_vals) == 0:
 94            raise ValueError("init_values must not be empty")
 95
 96        init_special_arr: list[SpecialValue | None] = []
 97        finite_init_values: list[float] = []
 98
 99        for i, v in enumerate(init_vals):
100            sv = normalize_special_value(v)
101            init_special_arr.append(sv)
102            if sv is None:
103                fv = float(v)
104                if not (min_val <= fv <= max_val):
105                    raise ValueError(
106                        f"init_values must all lie within [{min_val}, {max_val}], "
107                        f"but found value {fv} at index {i}"
108                    )
109                finite_init_values.append(fv)
110            else:
111                if sv == "+inf" and max_limit is not None:
112                    raise ValueError(f"init_values[{i}] cannot be +inf when max_limit is set")
113                if sv == "-inf" and min_limit is not None:
114                    raise ValueError(f"init_values[{i}] cannot be -inf when min_limit is set")
115                finite_init_values.append(max_val if sv == "+inf" else min_val)
116
117        self._arr_length = len(finite_init_values)
118        self._current_idx = init_idx % self._arr_length
119        self._current_arr = np.asarray(finite_init_values.copy())
120        self._default_arr = np.asarray(finite_init_values.copy())
121        self._current_special_arr = init_special_arr.copy()
122
123        init_val = init_special_arr[self._current_idx]
124        if init_val is None:
125            init_val = finite_init_values[self._current_idx]
126
127        super().__init__(
128            label=label,
129            unit=unit,
130            min_val=min_val,
131            max_val=max_val,
132            steps=steps,
133            init_val=init_val,
134            val_fmt=val_fmt,
135            min_limit=min_limit,
136            max_limit=max_limit,
137            min_limit_inclusive=min_limit_inclusive,
138            max_limit_inclusive=max_limit_inclusive,
139            show_range=show_range,
140            orientation=orientation,
141            editable=editable,
142            parent=parent,
143        )
144
145        self.html_label = html_label
146        self._use_same_values = False
147
148        self._right_click_items.append({"id": "sep"})
149        self._right_click_items.append({"id": "show_val", "text": "Show Values..."})
150
151        self.set_arr_values(self._current_arr)
152        self.set_index(self._current_idx)
153        self.set_use_same_values(use_same_values)
154
155        self.valueChanged.connect(self._on_single_value_changed)
156        self.defaultChanged.connect(self._on_single_default_changed)

197    def _on_single_value_changed(self, new_value) -> None:
198        """Update the stored current-value array after the active value changes.
199
200        Args:
201            new_value: New active slider value. May be finite or a special
202                infinity marker.
203        """
204        sv = normalize_special_value(new_value)
205        if sv is None:
206            finite_value = float(new_value)
207            self._current_arr[self._current_idx] = finite_value
208            if self._use_same_values:
209                self._current_arr[:] = finite_value
210
211        self._current_special_arr[self._current_idx] = sv
212        if self._use_same_values:
213            self._current_special_arr[:] = [sv] * self._arr_length
214
215        self.arrayChanged.emit(self.get_arr_values())

243    def _open_edit_config_dlg(self) -> None:
244        """Open and apply the configuration dialog for range and display settings.
245
246        On acceptance, this updates the stored format, numeric range, and
247        show-range state. If the new range would exclude current or default array
248        values, the user is asked whether those values should be clamped.
249        """
250        dlg = EditConfigDialog(
251            slider_name=self._label_text,
252            value_is_int=self.value_is_int,
253            min_val=self._min_value,
254            max_val=self._max_value,
255            val_fmt=self._val_fmt,
256            show_range=self._show_range,
257            min_limit=self._min_limit,
258            max_limit=self._max_limit,
259            min_limit_inclusive=self._min_limit_inclusive,
260            max_limit_inclusive=self._max_limit_inclusive,
261        )
262        if dlg.exec() == QDialog.Accepted:
263            new_min, new_max, new_fmt, new_show_range = dlg.get_values()
264
265            old_fmt = self._val_fmt
266            old_show_range = self._show_range
267            old_min = self._min_value
268            old_max = self._max_value
269
270            self.set_format(new_fmt)
271            new_min = self.evaluate_value(new_min)
272            new_max = self.evaluate_value(new_max)
273            cur_min = np.min(self._current_arr)
274            cur_max = np.max(self._current_arr)
275            dft_min = np.min(self._default_arr)
276            dft_max = np.max(self._default_arr)
277
278            if cur_min < new_min or dft_min < new_min or cur_max > new_max or dft_max > new_max:
279                result = multi_slider_ask_clamp_value(
280                    proposed_range=f"[{new_min}, {new_max}]",
281                    cur_min=self._value_to_text(cur_min),
282                    cur_max=self._value_to_text(cur_max),
283                    dft_min=self._value_to_text(dft_min),
284                    dft_max=self._value_to_text(dft_max),
285                    parent=self,
286                )
287                if result != AskResult.YES:
288                    self.set_format(old_fmt)
289                    return
290
291            self._show_range = new_show_range
292            self.set_range(new_min, new_max)
293            if (
294                old_min != new_min
295                or old_max != new_max
296                or old_fmt != new_fmt
297                or old_show_range != new_show_range
298            ):
299                self.configChanged.emit(self.get_config())

389    def set_format(self, val_fmt: str) -> None:
390        """Update the display format and re-evaluate stored values.
391
392        Args:
393            val_fmt: New display format string.
394
395        Note:
396            All stored current and default finite values are re-evaluated using
397            the new format. ``arrayChanged`` is emitted if any current value
398            changes as a result.
399        """
400        super().set_format(val_fmt)
401        cur_values_changed = False
402        for i in range(self._arr_length):
403            old_cur = self._current_arr[i]
404            old_dft = self._default_arr[i]
405            new_cur = self._current_arr[i] = self.evaluate_value(old_cur)
406            new_dft = self._default_arr[i] = self.evaluate_value(old_dft)
407            if new_cur != old_cur:
408                cur_values_changed = True
409        if cur_values_changed:
410            self.arrayChanged.emit(self._current_arr.copy())

445    def set_arr_values(
446        self,
447        values: np.ndarray,
448        idx: int = 0,
449        default_preserve_mode: Literal["none", "first", "all"] = "none",
450        same_values_mode: Literal["ignore", "apply"] = "ignore",
451    ) -> None:
452        """Replace the stored current values array.
453
454        Args:
455            values: New array of values. Must be non-empty, and each entry must
456                satisfy the slider's hard-limit constraints.
457            idx: Index to select after updating the array. Wrapped modulo the new
458                array length.
459            default_preserve_mode: Policy for preserving old default values.
460
461                - ``"none"``: Reset all default values from ``values``.
462                - ``"first"``: Preserve only the previous default at index 0.
463                - ``"all"``: Preserve all previous defaults where possible.
464            same_values_mode: Policy controlling interaction with same-values
465                mode.
466
467                - ``"ignore"``: Store ``values`` as provided.
468                - ``"apply"``: If same-values mode is enabled, force all current
469                  and default entries to match the selected entry.
470
471        Note:
472            If the new values extend beyond the current editable range, the
473            stored minimum and/or maximum range is expanded to include them, and
474            ``configChanged`` is emitted.
475
476        Raises:
477            ValueError: If ``values`` is empty, contains entries outside the
478                allowed hard limits, or if either mode argument is invalid.
479        """
480        N = len(values)
481        if N == 0:
482            raise ValueError("values must not be empty")
483
484        for i in range(N):
485            if not self.value_in_limit(values[i]):
486                raise ValueError(self._range_err_msg(f"value_{i}"))
487
488        if default_preserve_mode not in {"none", "first", "all"}:
489            raise ValueError(
490                "default_preserve_mode must be one of: 'none', 'first', 'all'"
491            )
492
493        if same_values_mode not in {"ignore", "apply"}:
494            raise ValueError(
495                "same_values_mode must be one of: 'ignore', 'apply'"
496            )
497
498        old_default_arr = self._default_arr.copy()
499        selected_idx = idx % N
500
501        self._arr_length = N
502        new_current = []
503        new_special = []
504
505        for i, v in enumerate(values):
506            sv = normalize_special_value(v)
507            if sv is None:
508                fv = self.evaluate_value(v)
509                if not self.value_in_limit(fv):
510                    raise ValueError(self._range_err_msg(f"value_{i}"))
511                new_current.append(fv)
512            else:
513                if sv == "+inf" and self._max_limit is not None:
514                    raise ValueError(self._range_err_msg(f"value_{i}"))
515                if sv == "-inf" and self._min_limit is not None:
516                    raise ValueError(self._range_err_msg(f"value_{i}"))
517                new_current.append(self._max_value if sv == "+inf" else self._min_value)
518
519            new_special.append(sv)
520
521        self._current_arr = np.asarray(new_current, dtype=float)
522        self._default_arr = self._current_arr.copy()
523        self._current_special_arr = new_special.copy()
524
525        if default_preserve_mode == "first":
526            if len(old_default_arr) > 0:
527                self._default_arr[0] = old_default_arr[0]
528        elif default_preserve_mode == "all":
529            n_copy = min(len(old_default_arr), N)
530            self._default_arr[:n_copy] = old_default_arr[:n_copy]
531
532        if same_values_mode == "apply" and self._use_same_values:
533            self._current_arr[:] = self._current_arr[selected_idx]
534            self._default_arr[:] = self._default_arr[selected_idx]
535
536        min_values = np.min(self._current_arr)
537        max_values = np.max(self._current_arr)
538
539        min_val = self._min_value
540        max_val = self._max_value
541        extended = False
542
543        if min_values < min_val:
544            min_val = min_values
545            extended = True
546        if max_values > max_val:
547            max_val = max_values
548            extended = True
549
550        self._min_value = min_val
551        self._max_value = max_val
552        self.set_index(selected_idx)
553
554        if extended:
555            self.configChanged.emit(self.get_config())

557    def set_range(self, min_val: float, max_val: float) -> None:
558        """Set the editable numeric range and clamp stored values if needed.
559
560        Args:
561            min_val: New minimum value.
562            max_val: New maximum value.
563
564        Note:
565            Both the current and default arrays are clamped to the new range. The
566            underlying slider, labels, and layout are refreshed. ``arrayChanged``
567            is emitted if any current value is clamped.
568
569        Raises:
570            ValueError: If ``min_val >= max_val`` or either bound violates the
571                hard-limit constraints.
572        """
573        new_min_val, new_max_val = self._validate_range(min_val, max_val)
574        self._min_value = new_min_val
575        self._max_value = new_max_val
576        cur_clamped = False
577        for i in range(self._arr_length):
578            cur_val = self._current_arr[i]
579            dft_val = self._default_arr[i]
580            if cur_val < new_min_val:
581                self._current_arr[i] = new_min_val
582                cur_clamped = True
583            elif cur_val > new_max_val:
584                self._current_arr[i] = new_max_val
585                cur_clamped = True
586            if dft_val < new_min_val:
587                self._default_arr[i] = new_min_val
588            elif dft_val > new_max_val:
589                self._default_arr[i] = new_max_val
590
591        self._current_value = self._current_arr[self._current_idx]
592        self._default_value = self._default_arr[self._current_idx]
593
594        self._update_mapping_anchors()
595        self._apply_range_to_slider()
596        self.blockSignals(True)
597        try:
598            self._slider.setValue(self._value_to_pos(self._current_value))
599        finally:
600            self.blockSignals(False)
601        self._update_all_labels()
602        self._update_value_width()
603        self._update_layout()
604
605        if cur_clamped:
606            self.arrayChanged.emit(self._current_arr.copy())

631    def set_arr_length(self, arr_length: int) -> None:
632        """Resize the stored arrays.
633
634        Args:
635            arr_length: New array length. Must be greater than zero.
636
637        Note:
638            When extending the arrays, new entries are filled with the value and
639            default value from index 0. When shrinking, trailing entries are
640            discarded. If the current index becomes invalid after shrinking, it
641            is reset to 0.
642
643        Raises:
644            ValueError: If ``arr_length <= 0``.
645        """
646        if arr_length <= 0:
647            raise ValueError("arr_length must be > 0")
648
649        old_length = self._arr_length

655            fill_dft = self._default_arr[0]

660                self._current_arr,
661                np.full(n_add, fill_cur, dtype=self._current_arr.dtype),

663            self._default_arr = np.concatenate([
664                self._default_arr,
665                np.full(n_add, fill_dft, dtype=self._default_arr.dtype),
666            ])
667            self._current_special_arr.extend([fill_cur_sv] * n_add)
668        else:
669            self._current_arr = self._current_arr[:arr_length]
670            self._default_arr = self._default_arr[:arr_length]
671            self._current_special_arr = self._current_special_arr[:arr_length]
672
673        self._arr_length = arr_length
674        if self._current_idx >= self._arr_length:
675            self._current_idx = 0
676        self._update_slider_for_current_idx()
677        self.arrayChanged.emit(self._current_arr.copy())

446    def set_arr_values(
447        self,
448        values: np.ndarray,
449        idx: int = 0,
450        default_preserve_mode: Literal["none", "first", "all"] = "none",
451        same_values_mode: Literal["ignore", "apply"] = "ignore",
452    ) -> None:
453        """Replace the stored current values array.
454
455        Args:
456            values: New array of values. Must be non-empty, and each entry must
457                satisfy the slider's hard-limit constraints.
458            idx: Index to select after updating the array. Wrapped modulo the new
459                array length.
460            default_preserve_mode: Policy for preserving old default values.
461
462                - ``"none"``: Reset all default values from ``values``.
463                - ``"first"``: Preserve only the previous default at index 0.
464                - ``"all"``: Preserve all previous defaults where possible.
465            same_values_mode: Policy controlling interaction with same-values
466                mode.
467
468                - ``"ignore"``: Store ``values`` as provided.
469                - ``"apply"``: If same-values mode is enabled, force all current
470                  and default entries to match the selected entry.
471
472        Note:
473            If the new values extend beyond the current editable range, the
474            stored minimum and/or maximum range is expanded to include them, and
475            ``configChanged`` is emitted.
476
477        Raises:
478            ValueError: If ``values`` is empty, contains entries outside the
479                allowed hard limits, or if either mode argument is invalid.
480        """
481        N = len(values)
482        if N == 0:
483            raise ValueError("values must not be empty")
484
485        for i in range(N):
486            if not self.value_in_limit(values[i]):
487                raise ValueError(self._range_err_msg(f"value_{i}"))
488
489        if default_preserve_mode not in {"none", "first", "all"}:
490            raise ValueError(
491                "default_preserve_mode must be one of: 'none', 'first', 'all'"
492            )
493
494        if same_values_mode not in {"ignore", "apply"}:
495            raise ValueError(
496                "same_values_mode must be one of: 'ignore', 'apply'"
497            )
498
499        old_default_arr = self._default_arr.copy()
500        selected_idx = idx % N
501
502        self._arr_length = N
503        new_current = []
504        new_special = []
505
506        for i, v in enumerate(values):
507            sv = normalize_special_value(v)
508            if sv is None:
509                fv = self.evaluate_value(v)
510                if not self.value_in_limit(fv):
511                    raise ValueError(self._range_err_msg(f"value_{i}"))
512                new_current.append(fv)
513            else:
514                if sv == "+inf" and self._max_limit is not None:
515                    raise ValueError(self._range_err_msg(f"value_{i}"))
516                if sv == "-inf" and self._min_limit is not None:
517                    raise ValueError(self._range_err_msg(f"value_{i}"))
518                new_current.append(self._max_value if sv == "+inf" else self._min_value)
519
520            new_special.append(sv)
521
522        self._current_arr = np.asarray(new_current, dtype=float)
523        self._default_arr = self._current_arr.copy()
524        self._current_special_arr = new_special.copy()
525
526        if default_preserve_mode == "first":
527            if len(old_default_arr) > 0:
528                self._default_arr[0] = old_default_arr[0]
529        elif default_preserve_mode == "all":
530            n_copy = min(len(old_default_arr), N)
531            self._default_arr[:n_copy] = old_default_arr[:n_copy]
532
533        if same_values_mode == "apply" and self._use_same_values:
534            self._current_arr[:] = self._current_arr[selected_idx]
535            self._default_arr[:] = self._default_arr[selected_idx]
536
537        min_values = np.min(self._current_arr)
538        max_values = np.max(self._current_arr)
539
540        min_val = self._min_value
541        max_val = self._max_value
542        extended = False
543
544        if min_values < min_val:
545            min_val = min_values
546            extended = True
547        if max_values > max_val:
548            max_val = max_values
549            extended = True
550
551        self._min_value = min_val
552        self._max_value = max_val
553        self.set_index(selected_idx)
554
555        if extended:
556            self.configChanged.emit(self.get_config())

559        """Set the editable numeric range and clamp stored values if needed.
560
