 87def parse_data_filename(filename: str) -> dict[str, Any]:
 88    """Parse a light-curve filename.
 89
 90    Supported examples include:
 91
 92    - ``Source_v=12.3.txt``
 93    - ``Source_p1_v=123.1.txt``
 94
 95    Args:
 96        filename: File name to parse.
 97
 98    Returns:
 99        Mapping containing ``source``, ``period``, and ``velocity``.
100
101    Raises:
102        ValueError: If ``filename`` does not match the expected pattern.
103    """
104    pattern = re.compile(
105        r"^(?P<source>.+?)(?:_(?P<period>p\d+))?_v=(?P<velocity>[-+]?\d*\.?\d+)\.txt$"
106    )
107
108    match = pattern.match(filename)
109    if match is None:
110        raise ValueError(f"Invalid filename format: {filename}")
111
112    return {
113        "source": match.group("source"),
114        "period": match.group("period"),
115        "velocity": match.group("velocity"),
116    }

146def analyze_folder(
147    folder_path: Path,
148    parent: Optional[QWidget] = None,
149) -> Optional[DataFolderInfo]:
150    """Scan a source folder and summarize the files it contains.
151
152    Args:
153        folder_path: Folder to analyze.
154        parent: Optional parent widget for warning dialogs.
155
156    Returns:
157        Summary information for the folder, or ``None`` if the folder contents
158        are invalid for import.
159    """
160
161    folder_path = Path(folder_path)
162    txt_files = sorted(folder_path.glob("*.txt"))
163    json_files = sorted(folder_path.glob("*.json"))
164
165    if not txt_files:
166        show_warning(
167            "No text files found",
168            f"No .txt files were found in:\n{folder_path}",
169            parent=parent
170        )
171        return None
172
173    source_names: set[str] = set()
174    period_velocities: dict[str, list[str]] = {}
175    no_period_velocities: list[str] = []
176
177    for txt_path in txt_files:
178        try:
179            info = parse_data_filename(txt_path.name)
180        except ValueError:
181            continue
182
183        source_names.add(info["source"])
184        period = info["period"]
185        velocity = info["velocity"]
186
187        if period is None:
188            no_period_velocities.append(velocity)
189        else:
190            period_velocities.setdefault(period, []).append(velocity)
191
192    if not source_names:
193        show_warning(
194            "Invalid file names",
195            "No valid data files were found with the expected naming pattern.",
196            parent=parent
197        )
198        return None
199
200    if len(source_names) > 1:
201        show_warning(
202            "Multiple sources found",
203            f"More than one source name was found in this folder:\n{sorted(source_names)}",
204            parent=parent
205        )
206        return None
207
208    if period_velocities and no_period_velocities:
209        show_warning(
210            "Mixed file types",
211            "The folder contains both period-tagged and non-period-tagged data files.",
212            parent=parent
213        )
214        return None
215
216    source = next(iter(source_names))
217
218    for p in period_velocities:
219        period_velocities[p].sort(key=float)
220    no_period_velocities.sort(key=float)
221
222    params_no_period = False
223    params_by_period: dict[str, bool] = {}
224
225    for json_path in json_files:
226        try:
227            info = parse_params_filename(json_path.name)
228        except ValueError:
229            continue
230
231        if info["source"] != source:
232            continue
233
234        period = info["period"]
235        if period is None:
236            params_no_period = True
237        else:
238            params_by_period[period] = True
239
240    if period_velocities:
241        periods: dict[str, PeriodInfo] = {}
242        for period, velocities in period_velocities.items():
243            periods[period] = PeriodInfo(
244                n_velocities=len(velocities),
245                has_params=params_by_period.get(period, False),
246            )
247
248        return DataFolderInfo(
249            source=source,
250            path=folder_path,
251            has_periods=True,
252            periods=periods,
253            no_period_info=None,
254        )
255
256    return DataFolderInfo(
257        source=source,
258        path=folder_path,
259        has_periods=False,
260        periods={},
261        no_period_info=PeriodInfo(
262            n_velocities=len(no_period_velocities),
263            has_params=params_no_period,
264        ),
265    )

268def select_period_if_needed(
269    folder_info: DataFolderInfo,
270    parent: Optional[QWidget] = None,
271) -> Optional[str]:
272    """Return a selected period label when period-tagged data exist.
273
274    Args:
275        folder_info: Summary information for the scanned folder.
276        parent: Optional parent widget for the selection dialog.
277
278    Returns:
279        Selected period label, or ``None`` when no period is needed or the user
280        cancels the selection dialog.
281    """
282    if not folder_info.has_periods:
283        return None
284
285    periods_keys = sorted(
286        folder_info.periods.keys(),
287        key=lambda s: int(s[1:])
288    )
289
290    if len(periods_keys) == 1:
291        return periods_keys[0]
292
293    dlg = SelectPeriodDialog(
294        source=folder_info.source,
295        periods_keys=periods_keys,
296        periods=folder_info.periods,
297        parent=parent,
298    )
299    if dlg.exec() == QDialog.Accepted:
300        return dlg.selected_period()
301
302    return None

305def load_velocity_files(
306    folder: Path,
307    selected_period: Optional[str] = None,
308) -> dict[str, LightCurve]:
309    """Load all matching light-curve files from a source folder.
310
311    Args:
312        folder: Folder containing the text files.
313        selected_period: Period label to filter by. Use ``None`` for files
314            without period tags.
315
316    Returns:
317        Mapping from velocity labels to loaded light curves.
318    """
319
320    folder = Path(folder)
321    all_v: dict[str, LightCurve] = {}
322
323    for txt_path in sorted(folder.glob("*.txt")):
324        try:
325            meta = parse_data_filename(txt_path.name)
326        except ValueError:
327            continue
328
329        if meta["period"] != selected_period:
330            continue
331
332        time_vals = []
333        flux_vals = []
334
335        with txt_path.open("r", encoding="utf-8") as f:
336            for line in f:
337                line = line.strip()
338
339                if not line or line.startswith("#"):
340                    continue
341
342                parts = line.split()
343                if len(parts) < 2:
344                    continue
345
346                try:
347                    time = float(parts[0])
348                    flux = float(parts[1])
349                except ValueError:
350                    continue
351
352                time_vals.append(time)
353                flux_vals.append(flux)
354
355        v = meta["velocity"]
356        all_v[v] = LightCurve(
357            time=np.array(time_vals, dtype=float),
358            flux=np.array(flux_vals, dtype=float),
359        )
360
361    return all_v

364def load_params_file(
365    folder: Path,
366    source: str,
367    selected_period: Optional[str] = None,
368) -> Optional[dict[str, Any]]:
369    """Load a parameter JSON file for the selected source and period.
370
371    Args:
372        folder: Folder containing the parameter file.
373        source: Source name used in the file name.
374        selected_period: Optional period label.
375
376    Returns:
377        Parsed JSON content with restored special floats, or ``None`` if the
378        parameter file does not exist.
379    """
380    folder = Path(folder)
381
382    if selected_period is None:
383        filename = f"{source}_params.json"
384    else:
385        filename = f"{source}_{selected_period}_params.json"
386
387    params_path = folder / filename
388    if not params_path.exists():
389        return None
390
391    with params_path.open("r", encoding="utf-8") as f:
392        return restore_special_floats(json.load(f))

395def show_data_folder_summary(
396    folder_info: DataFolderInfo,
397    selected_period: Optional[str] = None,
398    parent: Optional[QWidget] = None,
399) -> None:
400    """Display a summary dialog for the selected data folder.
401
402    Args:
403        folder_info: Summary information for the folder.
404        selected_period: Optional selected period label.
405        parent: Optional parent widget for the summary dialog.
406    """
407
408    if selected_period is None:
409        if folder_info.no_period_info is None:
410            return
411        n_velocities = folder_info.no_period_info.n_velocities
412        has_params = folder_info.no_period_info.has_params
413    else:
414        period_info = folder_info.periods[selected_period]
415        n_velocities = period_info.n_velocities
416        has_params = period_info.has_params
417
418    show_folder_summary(
419        source=folder_info.source,
420        n_velocities=n_velocities,
421        has_params=has_params,
422        selected_period=selected_period,
423        parent=parent,
424    )

427def import_data_folder(
428    parent: Optional[QWidget] = None,
429) -> ImportedDataFolder:
430    """Import one data folder together with any matching saved parameters.
431
432    Args:
433        parent: Optional parent widget for dialogs.
434
435    Returns:
436        Bundle containing folder metadata, selected period, loaded light curves,
437        and any loaded parameter snapshot.
438    """
439
440    folder = select_folder(parent)
441    if folder is None:
442        return ImportedDataFolder(None, None, None, None)
443
444    folder_info = analyze_folder(folder, parent)
445    if folder_info is None:
446        return ImportedDataFolder(None, None, None, None)
447
448    if folder_info.has_periods:
449        selected_period = select_period_if_needed(folder_info, parent)
450        if selected_period is None:
451            return ImportedDataFolder(folder_info, None, None, None)
452    else:
453        selected_period = None
454
455    show_data_folder_summary(
456        folder_info=folder_info,
457        selected_period=selected_period,
458        parent=parent,
459    )
460
461    all_v = load_velocity_files(folder, selected_period)
462    params = load_params_file(
463        folder=folder,
464        source=folder_info.source,
465        selected_period=selected_period,
466    )
467
468    return ImportedDataFolder(
469        folder_info=folder_info,
470        selected_period=selected_period,
471        data=all_v,
472        params=params,
473    )

476def _demo_main() -> int:
477    """Run the data-import helpers as a small standalone demo.
478
479    Returns:
480        Process exit code.
481    """
482
483    from settings.app_style import set_app_style
484
485    app = QApplication([])
486    set_app_style(app)
487    result = import_data_folder()
488
489    if result.folder_info is not None and result.data is not None:
490        print("SourceSelection name:", result.folder_info.source)
491        print("Has periods:", result.folder_info.has_periods)
492        print("Selected period:", result.selected_period)
493
494        print("Velocities:")
495        for v in sorted(result.data, key=float):
496            print(v, len(result.data[v].time))
497
498        print("Params:", result.params)
499    return 0

def load_velocity_files( folder: pathlib.Path, selected_period: Optional[str] = None) -> dict[str, LightCurve]: 
306def load_velocity_files(
307    folder: Path,
308    selected_period: Optional[str] = None,
309) -> dict[str, LightCurve]:
310    """Load all matching light-curve files from a source folder.
311
312    Args:
313        folder: Folder containing the text files.
314        selected_period: Period label to filter by. Use ``None`` for files
315            without period tags.
316
317    Returns:
318        Mapping from velocity labels to loaded light curves.
319    """
320
321    folder = Path(folder)
322    all_v: dict[str, LightCurve] = {}
323
324    for txt_path in sorted(folder.glob("*.txt")):
325        try:
326            meta = parse_data_filename(txt_path.name)
327        except ValueError:
328            continue
329
330        if meta["period"] != selected_period:
331            continue
332
333        time_vals = []
334        flux_vals = []
335
336        with txt_path.open("r", encoding="utf-8") as f:
337            for line in f:
338                line = line.strip()
339
340                if not line or line.startswith("#"):
341                    continue
342
343                parts = line.split()
344                if len(parts) < 2:
345                    continue
346
347                try:
348                    time = float(parts[0])
349                    flux = float(parts[1])
350                except ValueError:
351                    continue
352
353                time_vals.append(time)
354                flux_vals.append(flux)
355
356        v = meta["velocity"]
357        all_v[v] = LightCurve(
358            time=np.array(time_vals, dtype=float),
359            flux=np.array(flux_vals, dtype=float),
360        )
361
362    return all_v
