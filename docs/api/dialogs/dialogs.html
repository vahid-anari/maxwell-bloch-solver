  66def _try_load_app_icon_pixmap(size: int) -> Optional[QPixmap]:
  67    """Return the application icon pixmap when the icon file can be loaded.
  68
  69    Args:
  70        size: Requested icon size in pixels.
  71
  72    Returns:
  73        Scaled pixmap if the icon file exists and loads successfully, otherwise
  74        ``None``.
  75    """
  76
  77    icon_path = Path(APP_ICON_PATH)
  78    if not icon_path.exists():
  79        return None
  80
  81    pixmap = QPixmap(str(icon_path))
  82    if pixmap.isNull():
  83        return None
  84
  85    return pixmap.scaled(
  86        size,
  87        size,
  88        Qt.KeepAspectRatio,
  89        Qt.SmoothTransformation,
  90    )

 123class TitledDialogBase(QDialog):
 124    """Common dialog base class with title text, optional icon, and body helpers."""
 125
 126    def __init__(
 127        self,
 128        title: str,
 129        text: str,
 130        informative_text: str = "",
 131        icon: Optional[DialogIcon] = None,
 132        parent: Optional[QWidget] = None,
 133    ) -> None:
 134        """Initialize the common titled-dialog layout.
 135
 136        Args:
 137            title: Window title.
 138            text: Main dialog text.
 139            informative_text: Optional secondary explanatory text.
 140            icon: Optional standard dialog icon preset.
 141            parent: Optional parent widget.
 142        """
 143
 144        super().__init__(parent)
 145        _set_window_icon_from_app_icon(self)
 146
 147        left_margin = 5
 148        icon_margin = 10
 149        right_margin = 8
 150        top_margin = 0
 151        bottom_margin = 7
 152
 153        self.setWindowTitle(title)
 154
 155        text_layout = QVBoxLayout()
 156        text_layout.setSpacing(6)
 157
 158        text_label = _make_html_label(text.replace("\n", "<br>"))
 159        text_layout.addWidget(text_label)
 160
 161        if informative_text:
 162            info_label = _make_html_label(informative_text.replace("\n", "<br>"))
 163            text_layout.addWidget(info_label)
 164
 165        row = QHBoxLayout()
 166        row.addSpacing(left_margin)
 167        if icon is not None:
 168            std_icon = self.style().standardIcon(icon.value)
 169            pixmap = std_icon.pixmap(32, 32)
 170            icon_label = QLabel()
 171            icon_label.setPixmap(pixmap)
 172            icon_label.setFixedSize(icon_label.sizeHint())
 173            row.addWidget(icon_label, alignment=Qt.AlignTop)
 174            row.addSpacing(icon_margin)
 175        row.addLayout(text_layout)
 176        row.addSpacing(right_margin)
 177
 178        self._layout = QVBoxLayout(self)
 179        self._layout.addSpacing(top_margin)
 180        self._layout.addLayout(row)
 181        self._layout.addSpacing(bottom_margin)
 182        self._layout.addStretch(1)
 183
 184    def add_layout(self, layout: QLayout) -> None:
 185        """Add a layout to the dialog body.
 186
 187        Args:
 188            layout: Layout to append to the main dialog layout.
 189        """
 190
 191        self._layout.addLayout(layout)
 192
 193    def add_widget(self, widget: QWidget) -> None:
 194        """Add a widget to the dialog body.
 195
 196        Args:
 197            widget: Widget to append to the main dialog layout.
 198        """
 199
 200        self._layout.addWidget(widget)
 201
 202    def add_stretch(self, stretch: int = 1) -> None:
 203        """Add stretch space to the dialog body.
 204
 205        Args:
 206            stretch: Stretch factor to add.
 207        """
 208
 209        self._layout.addStretch(stretch)

 255class InfoDialog(QDialog):
 256    """Rich-text dialog for longer informational content."""
 257
 258    def __init__(
 259        self,
 260        title: str,
 261        html_text: str,
 262        icon: Optional[DialogIcon] = DialogIcon.INFORMATION,
 263        parent: Optional[QWidget] = None,
 264        copy_button: bool = True,
 265    ) -> None:
 266        """Initialize the rich informational dialog.
 267
 268        Args:
 269            title: Window title.
 270            html_text: Rich HTML content shown in the text browser.
 271            icon: Optional standard dialog icon preset.
 272            parent: Optional parent widget.
 273            copy_button: Whether to include a button that copies the visible text.
 274        """
 275
 276        super().__init__(parent)
 277        _set_window_icon_from_app_icon(self)
 278
 279        self.setWindowTitle(title)
 280        self.resize(560, 360)
 281        self.setMinimumSize(420, 260)
 282
 283        main_layout = QVBoxLayout(self)
 284        main_layout.setContentsMargins(18, 16, 18, 16)
 285        main_layout.setSpacing(12)
 286
 287        top_row = QHBoxLayout()
 288        top_row.setSpacing(12)
 289        if icon is not None:
 290            std_icon = self.style().standardIcon(icon.value)
 291            pixmap = std_icon.pixmap(32, 32)
 292            icon_label = QLabel()
 293            icon_label.setPixmap(pixmap)
 294            icon_label.setFixedSize(icon_label.sizeHint())
 295            top_row.addWidget(icon_label, alignment=Qt.AlignTop)
 296
 297        title_label = _make_html_label(f"<b>{html.escape(title)}</b>")
 298        top_row.addWidget(title_label, 1, alignment=Qt.AlignVCenter)
 299        main_layout.addLayout(top_row)
 300
 301        browser = QTextBrowser(self)
 302        browser.setOpenExternalLinks(True)
 303        browser.setReadOnly(True)
 304        browser.setTextInteractionFlags(Qt.TextBrowserInteraction)
 305        browser.setHtml(html_text)
 306        browser.moveCursor(QTextCursor.Start)
 307        self._browser = browser
 308        main_layout.addWidget(browser, 1)
 309
 310        btn_layout = QHBoxLayout()
 311        btn_layout.addStretch(1)
 312
 313        if copy_button:
 314            copy_btn = QPushButton("Copy")
 315            copy_btn.clicked.connect(self._copy_to_clipboard)
 316            btn_layout.addWidget(copy_btn)
 317
 318        close_btn = QPushButton("Close")
 319        close_btn.setDefault(True)
 320        close_btn.setAutoDefault(True)
 321        close_btn.clicked.connect(self.accept)
 322        btn_layout.addWidget(close_btn)
 323
 324        main_layout.addLayout(btn_layout)
 325
 326    def _copy_to_clipboard(self) -> None:
 327        """Copy the visible dialog text to the clipboard."""
 328
 329        QGuiApplication.clipboard().setText(self._browser.toPlainText())
