162    @staticmethod
163    def _wrap_report_lines(text: str, width: int = 110) -> list[str]:
164        """Wrap report text into fixed-width lines.
165
166        Args:
167            text: Full report text.
168            width: Target line width.
169
170        Returns:
171            Wrapped lines suitable for pagination.
172        """
173        wrapped: list[str] = []
174        for line in text.splitlines():
175            if not line.strip():
176                wrapped.append("")
177                continue
178            indent = len(line) - len(line.lstrip(" "))
179            prefix = " " * indent
180            body = line[indent:]
181            chunks = textwrap.wrap(
182                body,
183                width=max(20, width - indent),
184                replace_whitespace=False,
185                drop_whitespace=False,
186            )
187            if not chunks:
188                wrapped.append(prefix)
189            else:
190                wrapped.extend(prefix + chunk for chunk in chunks)
191        return wrapped

193    def _build_export_figure(self) -> plt.Figure:
194        """Build a standalone figure for export.
195
196        Returns:
197            Matplotlib figure containing the top panel and bottom-panel curves.
198        """
199        use_tex = USE_LATEX
200        params = self._get_params()
201        dc = self._get_displayed_curves()
202        fit_mode = self._get_fit_mode()
203
204        bottom_curves = {
205            "w": dc.results.w,
206            "lambda_n": dc.results.lambda_n,
207            "A0": dc.results.A0,
208        }
209        nrows = 1 + len(bottom_curves)
210        height_ratios = [3.0] + [1.0] * len(bottom_curves)
211
212        fig, axes = plt.subplots(
213            nrows=nrows,
214            ncols=1,
215            sharex=True,
216            figsize=(
217                8.27 - 2 * EXPORT_PLOT_PAD_INCHES,
218                11.69 - 2 * EXPORT_PLOT_PAD_INCHES,
219            ),
220            gridspec_kw={"height_ratios": height_ratios},
221            constrained_layout=True,
222        )
223        axes = [axes] if nrows == 1 else list(axes)
224
225        top_ax = axes[0]
226        data_style = self._canvas.get_curve_style("data_points") or {}
227        flux_style = self._canvas.get_curve_style("flux") or {}
228        top_ax.plot(dc.data.time, dc.data.flux, **self._style_kwargs(data_style))
229        top_ax.plot(dc.results.time, dc.results.intensity, **self._style_kwargs(flux_style))
230
231        t_limits = self._canvas.get_time_limits()
232        top_ax.set_xlim(t_limits)
233        top_ax.set_ylim(self._canvas.get_flux_limits())
234        self._canvas.add_ax_props(top_ax)
235
236        flux_label = "F" if fit_mode else "I"
237        flux_unit = params["data.unit.flux"] if fit_mode else "I_0"
238        top_ax.set_ylabel(self._axis_label(flux_label, flux_unit), usetex=use_tex)
239        self._apply_grid(top_ax, "time", "flux")
240
241        for ax, (curve_id, ys) in zip(axes[1:], bottom_curves.items()):
242            style = self._canvas.get_curve_style(curve_id) or {}
243            ax.plot(dc.results.time, ys, **self._style_kwargs(style))
244            self._canvas.add_ax_props(ax)
245            ax.set_ylabel(
246                self._axis_label(style.get("label", ""), style.get("unit", "")),
247                usetex=use_tex,
248            )
249            ax.set_ylim(get_range(
250                limits=params[f"display.range.{curve_id}"],
251                arr1=cut_y_by_x(
252                    y=getattr(dc.results, curve_id),
253                    x=dc.results.time,
254                    xmin=t_limits[0],
255                    xmax=t_limits[1],
256                ),
257            ))
258            self._apply_grid(ax, "time", "bottom")
259
260        time_unit = params["data.unit.time"] if fit_mode else "T_0"
261        axes[-1].set_xlabel(self._axis_label("t", time_unit), usetex=use_tex)
262        for ax in axes[:-1]:
263            ax.label_outer()
264
265        fig.show()
266        return fig

282    @staticmethod
283    def _style_kwargs(style: dict[str, Any]) -> dict[str, Any]:
284        """Convert a stored style mapping into ``plot`` keyword arguments.
285
286        Args:
287            style: Curve style mapping.
288
289        Returns:
290            Keyword-argument mapping for Matplotlib plotting.
291        """
292        out = {
293            "color": style["color"],
294            "linestyle": style["linestyle"],
295            "linewidth": style["linewidth"],
296        }
297        marker = style.get("marker")
298        if marker not in (None, "", "None", "none", " "):
299            out["marker"] = marker
300            out["markersize"] = style.get("markersize", 6.0)
301        return out

305        """Build a LaTeX axis label from a quantity label and unit.

307        Args:
308            label: Quantity label.
309            unit: Unit string.
310
311        Returns:

326    def __init__(
327        self,
328        figure_props: Dict[str, Any],
329        axes_props: Dict[str, Any],
330        lines_props: Dict[str, Any],
331        t_limits: Tuple[float, float],
332        parent=None,
333    ):
334        """Initialize the canvas from the JSON-driven plot configuration.
335
336        Args:
337            figure_props: Figure-level settings.
338            axes_props: Axis-level settings.
339            lines_props: Line-style specifications for all curves.
340            t_limits: Initial time-axis limits.
341            parent: Optional Qt parent widget.
342        """
343        self._use_tex = USE_LATEX
344        self._axes_props = axes_props
345
346        science_style = figure_props["science_style"]
347        fig_main_kws = figure_props["main"]
348        fig_margin_kws = figure_props["margin"]
349        self._n_axes = fig_main_kws["nrows"]
350        if science_style:
351            plt.style.use("science")
352
353        self._t_limits = t_limits
354        self._flux_limits = (-1, 1)
355        self._cosh_peak_lines = [None] * self._n_axes
356        self._cosh_peak_spans = [None] * self._n_axes
357
358        self._fig, self._axes = plt.subplots(**fig_main_kws)
359        self._fig.subplots_adjust(**fig_margin_kws)
360
361        top_panel = lines_props["top_panel"]
362        self._data_points_curve = self._make_curve(self._axes[0], top_panel["data_points"])
363        self._flux_curve = self._make_curve(self._axes[0], top_panel["flux"])
364
365        bottom_panel = lines_props["bottom_panel"]
366        self._bottom_panel_curves: Dict[str, PlotCurve] = {}
367        self._bottom_panel_labels: Dict[str, str] = {}
368        for name, props in bottom_panel.items():
369            curve = self._make_curve(self._axes[1], props)
370            self._bottom_panel_curves[name] = curve
371            self._bottom_panel_labels[name] = props.get("combo_label", "")
372
373        for i in range(self._n_axes):
374            ax = self._axes[i]
375            self.add_ax_props(ax)
376            self._cosh_peak_lines[i] = ax.axvline(0.0, **axes_props["cosh_peak"]["line"], visible=False)
377            self._cosh_peak_spans[i] = ax.axvspan(0.0, 1.0, **axes_props["cosh_peak"]["span"], visible=False)
378
379        super().__init__(self._fig)
380        self.setParent(parent)
381        self.setMinimumWidth(700)
382        self.setMinimumHeight(350)
