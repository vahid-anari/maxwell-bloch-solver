 314    def _can_leave(self) -> bool:
 315        """Return whether it is safe to discard the current state.
 316
 317        Shows an unsaved-changes dialog when modifications are pending so the
 318        user can save, discard, or cancel.
 319
 320        Returns:
 321            ``True`` if navigation can proceed, otherwise ``False``.
 322        """
 323        if not self._status_bar.is_modified():
 324            return True
 325
 326        if self._fit_mode:
 327            result = ask_question(
 328                "Unsaved changes",
 329                f"Velocity {self._velocity_bar.get_current_velocity()} has unsaved changes.",
 330                "Do you want to save them?",
 331                no_btn_label="No",
 332                yes_btn_label="Yes",
 333                cancel_btn_label="Cancel",
 334                parent=self,
 335            )
 336            if result == AskResult.CANCEL:
 337                return False
 338            if result == AskResult.YES:
 339                self._velocity_bar.save_parameters()
 340            elif result == AskResult.NO:
 341                self._velocity_bar.set_modified(False)
 342            return True
 343
 344        result = ask_question(
 345            "Unsaved changes",
 346            "Current parameters have unsaved changes.",
 347            "Do you want to save them?",
 348            no_btn_label="No",
 349            yes_btn_label="Yes",
 350            cancel_btn_label="Cancel",
 351            parent=self,
 352        )
 353        if result == AskResult.CANCEL:
 354            return False
 355        if result == AskResult.YES:
 356            self._save()
 357        return True

 456    # ---- Cosh peak ----
 457    def _update_cosh_peak(self) -> None:
 458        """Reposition and show or hide the cosh-peak marker on the canvas."""
 459        if self._cosh_peak_path is None:
 460            return
 461        if self._solver.is_solving:
 462            return
 463
 464        params = self._solver.current_params
 465        cosh_values = params[self._cosh_peak_path]
 466        symmetric = cosh_values["symmetric"]
 467        idx = cosh_values["current_idx"]
 468        amplitude = cosh_values["a"][idx]
 469        x0 = cosh_values["x0"][idx]
 470
 471        if amplitude == 0:
 472            wl = wr = 0.0
 473        elif symmetric:
 474            wl = wr = cosh_values["w"][idx] / 2.0
 475        else:
 476            wl = cosh_values["wl"][idx]
 477            wr = cosh_values["wr"][idx]
 478
 479        if self._fit_mode:
 480            from app.update_pipeline import get_time_unit_scale
 481            scale = get_time_unit_scale(params)
 482            x0 = params["results.offset.time"] + x0 * scale
 483            wl *= scale
 484            wr *= scale
 485
 486        self._canvas.set_cosh_peak_position(x0, wl, wr)
 487        cat = self._cosh_peak_path.split(".")[1]
 488        in_correct_tab = (
 489            (cat == "bcs" and self._bottom_plot == "A0")
 490            or (cat == "pump" and self._bottom_plot == "lambda_n")
 491        )
 492        self._canvas.set_cosh_peak_visible(self._show_cosh_peak and in_correct_tab)

 458    def _update_cosh_peak(self) -> None:
 459        """Reposition and show or hide the cosh-peak marker on the canvas."""
 460        if self._cosh_peak_path is None:
 461            return
 462        if self._solver.is_solving:
 463            return
 464
 465        params = self._solver.current_params
 466        cosh_values = params[self._cosh_peak_path]
 467        symmetric = cosh_values["symmetric"]
 468        idx = cosh_values["current_idx"]
 469        amplitude = cosh_values["a"][idx]
 470        x0 = cosh_values["x0"][idx]
 471
 472        if amplitude == 0:
 473            wl = wr = 0.0
 474        elif symmetric:
 475            wl = wr = cosh_values["w"][idx] / 2.0
 476        else:
 477            wl = cosh_values["wl"][idx]
 478            wr = cosh_values["wr"][idx]

 480        if self._fit_mode:

 483            x0 = params["results.offset.time"] + x0 * scale
 484            wl *= scale
 485            wr *= scale
 486
 487        self._canvas.set_cosh_peak_position(x0, wl, wr)
 488        cat = self._cosh_peak_path.split(".")[1]
 489        in_correct_tab = (
 490            (cat == "bcs" and self._bottom_plot == "A0")
