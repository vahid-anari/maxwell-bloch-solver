app.settings_manager

Application-level settings, view preferences, line colors, and recent folders.

This module owns all QSettings persistence for the main window: app-default parameter snapshots, view-preference toggles, per-curve line colors, and the recent-data-folder history.

 65class Settings:
 66    """Top-level settings container that aggregates all persistent state."""
 67
 68    def __init__(
 69        self,
 70        organization: str,
 71        application: str,
 72        canvas: "PlotCanvas",
 73        default_lines_config: Dict[str, Any],
 74    ) -> None:
 75        """Initialize the settings container.
 76
 77        Args:
 78            organization: QSettings organization string.
 79            application: QSettings application string.
 80            canvas: Plot canvas whose line colors are managed.
 81            default_lines_config: Factory-default line-configuration mapping.
 82        """
 83        self._settings = QSettings(organization, application)
 84        self.recent_folders = RecentFoldersManager(self._settings)
 85        self.line_colors = LineColorManager(
 86            canvas=canvas,
 87            settings=self._settings,
 88            default_lines_config=default_lines_config,
 89        )
 90
 91    def _serialize_for_settings(self, obj: Any) -> str:
 92        """Serialize an object to a JSON string for QSettings storage.
 93
 94        Args:
 95            obj: Object to serialize.
 96
 97        Returns:
 98            JSON string representation.
 99        """
100        return pretty_json(obj)
101
102    def _deserialize_from_settings(self, text: str) -> Any:
103        """Deserialize a JSON string from QSettings.
104
105        Args:
106            text: Stored JSON string.
107
108        Returns:
109            Deserialized object with special floats restored.
110        """
111        return restore_special_floats(json.loads(text))
112
113    def load_saved_app_defaults(self) -> tuple[dict, dict] | None:
114        """Load saved application defaults from QSettings.
115
116        Returns:
117            Tuple ``(config, params)`` if saved defaults exist, otherwise
118            ``None``.
119        """
120        config_text = self._settings.value(APP_DEFAULT_CONFIG_KEY, "", type=str)
121        params_text = self._settings.value(APP_DEFAULT_PARAMS_KEY, "", type=str)
122        if not config_text or not params_text:
123            return None
124        try:
125            return (
126                self._deserialize_from_settings(config_text),
127                self._deserialize_from_settings(params_text),
128            )
129        except Exception:
130            return None
131
132    def save_as_app_default(self, params_tab_widget: "ParameterTabsWidget") -> Tuple[bool, str]:
133        """Save current parameter values and config as new application defaults.
134
135        Args:
136            params_tab_widget: Widget whose state should be persisted.
137
138        Returns:
139            Tuple ``(success, message)`` where ``message`` is empty on success or
140            contains an error description on failure.
141        """
142        try:
143            self._settings.setValue(
144                APP_DEFAULT_CONFIG_KEY,
145                self._serialize_for_settings(params_tab_widget.get_config()),
146            )
147            self._settings.setValue(
148                APP_DEFAULT_PARAMS_KEY,
149                self._serialize_for_settings(params_tab_widget.get_value()),
150            )
151            self._settings.sync()
152            return True, ""
153        except Exception as exc:
154            return False, str(exc)
155
156    def get_view_preference(self, action_id: str, default: bool = False) -> bool:
157        """Return one persisted boolean view preference.
158
159        Args:
160            action_id: Menu-action identifier used to look up the settings key.
161            default: Value returned when the key is not set.
162
163        Returns:
164            Stored preference value, or ``default`` if absent.
165        """
166        key = VIEW_PREFERENCE_KEYS.get(action_id)
167        if key is None:
168            return default
169        value = self._settings.value(key, default)
170        if isinstance(value, bool):
171            return value
172        if isinstance(value, str):
173            return value.strip().lower() in {"1", "true", "yes", "on"}
174        return bool(value)
175
176    def set_view_preference(self, action_id: str, checked: bool) -> None:
177        """Persist one boolean view preference.
178
179        Args:
180            action_id: Menu-action identifier used to locate the settings key.
181            checked: Boolean state to persist.
182        """
183        key = VIEW_PREFERENCE_KEYS.get(action_id)
184        if key is not None:
185            self._settings.setValue(key, checked)

188class LineColorManager:
189    """Manage curve colors, their persistence, and related menu icons."""
190
191    def __init__(
192        self,
193        canvas: "PlotCanvas",
194        settings: "QSettings",
195        default_lines_config: Dict[str, Any],
196    ) -> None:
197        """Initialize the line-color manager.
198
199        Args:
200            canvas: Plot canvas whose curve colors are managed.
201            settings: Application QSettings instance.
202            default_lines_config: Factory-default line-configuration mapping.
203        """
204        self._canvas = canvas
205        self._settings = settings
206        self._default_lines_config = default_lines_config
207
208    def apply_saved_colors(self) -> None:
209        """Apply persisted colors to the canvas and redraw it."""
210
211        self._canvas.apply_curve_colors(self.load_saved_colors())
212        self._canvas.redraw()
213
214    def load_saved_colors(self) -> Dict[str, str]:
215        """Return a mapping of curve IDs to colors.
216
217        Returns:
218            Mapping containing default colors merged with any saved overrides.
219        """
220        colors = self._get_default_colors()
221        for curve_id in colors:
222            saved = self._settings.value(self._settings_key(curve_id), "", type=str)
223            if saved:
224                colors[curve_id] = saved
225        return colors
226
227    def choose_color(
228        self,
229        curve_id: str,
230        parent: "QWidget",
231        menu_bar: "MenuBarController",
232        on_combo_icons_changed: Callable[[], None],
233    ) -> None:
234        """Open a color dialog, then persist and apply the chosen color.
235
236        Args:
237            curve_id: Internal curve identifier.
238            parent: Parent widget for the color dialog.
239            menu_bar: Menu controller whose icons should be updated.
240            on_combo_icons_changed: Callback invoked after icon updates.
241        """
242        current = self._canvas.get_curve_color(curve_id) or "#000000"
243        color = QColorDialog.getColor(QColor(current), parent, "Choose Line Color")
244        if not color.isValid():
245            return
246
247        color_name = color.name()
248        if self._canvas.set_curve_color(curve_id, color_name):
249            self._save_color(curve_id, color_name)
250            self._update_menu_icon(menu_bar, curve_id, color_name)
251            on_combo_icons_changed()
252            self._canvas.redraw()
253
254    def reset_to_defaults(
255        self,
256        menu_bar: "MenuBarController",
257        on_combo_icons_changed: Callable[[], None],
258    ) -> None:
259        """Restore all curve colors to factory defaults.
260
261        Args:
262            menu_bar: Menu controller whose icons should be updated.
263            on_combo_icons_changed: Callback invoked after icon updates.
264        """
265        defaults = self._get_default_colors()
266        self._canvas.apply_curve_colors(defaults)
267        for curve_id, color in defaults.items():
268            self._save_color(curve_id, color)
269            self._update_menu_icon(menu_bar, curve_id, color)
270        on_combo_icons_changed()
271        self._canvas.redraw()
272
273    def update_all_menu_icons(self, menu_bar: "MenuBarController") -> None:
274        """Refresh every line-color menu icon to match saved colors.
275
276        Args:
277            menu_bar: Menu controller whose icons should be updated.
278        """
279        for curve_id, color in self.load_saved_colors().items():
280            self._update_menu_icon(menu_bar, curve_id, color)
281
282    def build_menu_items(self) -> List[Dict[str, Any]]:
283        """Build menu-item specifications for the line-colors submenu.
284
285        Returns:
286            List of menu-item specification dictionaries.
287        """
288        items: List[Dict[str, Any]] = [
289            {"id": f"{LINE_COLOR_ACTION_PREFIX}data_points", "text": "Data Points..."},
290            {"id": f"{LINE_COLOR_ACTION_PREFIX}flux", "text": "Flux..."},
291            {"id": "sep"},
292        ]
293        bottom_panel = self._default_lines_config.get("bottom_panel", {})
294        for name, props in bottom_panel.items():
295            text = props.get("combo_label") or props.get("label") or name
296            items.append({"id": f"{LINE_COLOR_ACTION_PREFIX}{name}", "text": f"{text}..."})
297
298        items.append({"id": "sep"})
299        items.append({"id": RESET_LINE_COLORS_ACTION_ID, "text": "Restore Default Line Colors"})
300        return items
301
302    @staticmethod
303    def make_color_icon(color: str, size: int = 12) -> QIcon:
304        """Return a square icon filled with a given color.
305
306        Args:
307            color: Color value.
308            size: Icon size in pixels.
309
310        Returns:
311            Qt icon filled with the requested color.
312        """
313        pix = QPixmap(size, size)
314        pix.fill(QColor(mcolors.to_hex(color)))
315        return QIcon(pix)
316
317    def _get_default_colors(self) -> Dict[str, str]:
318        """Extract factory-default curve colors from the line config.
319
320        Returns:
321            Mapping from curve identifiers to default colors.
322        """
323        lines = self._default_lines_config
324        out: Dict[str, str] = {}
325        top_panel = lines.get("top_panel", {})
326        for name in ("data_points", "flux"):
327            color = top_panel.get(name, {}).get("props", {}).get("color")
328            if color:
329                out[name] = str(color)
330        for name, item in lines.get("bottom_panel", {}).items():
331            color = item.get("props", {}).get("color")
332            if color:
333                out[name] = str(color)
334        return out
335
336    def _settings_key(self, curve_id: str) -> str:
337        """Return the QSettings key for a curve color.
338
339        Args:
340            curve_id: Internal curve identifier.
341
342        Returns:
343            Full QSettings key string.
344        """
345        return f"{LINE_COLOR_SETTINGS_PREFIX}{curve_id}"
346
347    def _save_color(self, curve_id: str, color: str) -> None:
348        """Persist one curve color.
349
350        Args:
351            curve_id: Internal curve identifier.
352            color: Color value to persist.
353        """
354        self._settings.setValue(self._settings_key(curve_id), color)
355
356    def _update_menu_icon(self, menu_bar: "MenuBarController", curve_id: str, color: str) -> None:
357        """Update the menu icon associated with one curve color.
358
359        Args:
360            menu_bar: Menu controller holding the relevant action.
361            curve_id: Internal curve identifier.
362            color: New color value.
363        """
364        action_id = f"{LINE_COLOR_ACTION_PREFIX}{curve_id}"
365        action = menu_bar._actions.get(action_id)
366        if action is not None:
367            action.setIcon(self.make_color_icon(color))
368            action.setIconVisibleInMenu(True)

371class RecentFoldersManager:
372    """Manage the recent-data-folder list and its menu representation."""
373
374    def __init__(self, settings: "QSettings") -> None:
375        """Initialize the recent-folders manager.
376
377        Args:
378            settings: Application QSettings instance used for persistence.
379        """
380        self._settings = settings
381        self._folders: list[str] = self._load()
382        self.action_map: dict[str, str] = {}
383
384    @property
385    def folders(self) -> List[str]:
386        """Return the current recent-folder list.
387
388        Returns:
389            Copy of the stored recent-folder list.
390        """
391        return list(self._folders)
392
393    def remember(self, folder_path: Path) -> None:
394        """Insert a folder at the front of the recent list and persist it.
395
396        Args:
397            folder_path: Folder path to remember.
398        """
399        path_str = str(folder_path.resolve())
400        items = [p for p in self._folders if p != path_str]
401        items.insert(0, path_str)
402        self._folders = items[:MAX_RECENT_DATA_FOLDERS]
403        self._save()
404
405    def remove(self, folder_path: str) -> None:
406        """Remove one folder from the recent list and persist it.
407
408        Args:
409            folder_path: Folder path to remove.
410        """
411        self._folders = [p for p in self._folders if p != folder_path]
412        self._save()
413
414    def clear(self) -> None:
415        """Clear the recent-folder list and persist it."""
416
417        self._folders = []
418        self._save()
419
420    def build_menu_items(self) -> List[dict[str, Any]]:
421        """Build the submenu specification for recent data folders.
422
423        Returns:
424            List of menu-item specification dictionaries.
425        """
426        self.action_map.clear()
427        items: list[dict[str, Any]] = []
428
429        if not self._folders:
430            items.append({
431                "id": "recent_data_folders_empty",
432                "text": "No Recent Folders",
433                "enabled": False,
434            })
435        else:
436            for index, folder_path in enumerate(self._folders):
437                action_id = f"{RECENT_FOLDER_ACTION_PREFIX}{index}"
438                self.action_map[action_id] = folder_path
439                items.append({
440                    "id": action_id,
441                    "text": self._format_label(folder_path),
442                    "enabled": True,
443                })
444            items.append({"id": "sep"})
445
446        items.append({
447            "id": "clear_recent_data_folders",
448            "text": "Clear Recent Data Folders",
449            "enabled": bool(self._folders),
450        })
451        return items
452
453    def _load(self) -> List[str]:
454        """Load and sanitize the recent-folder list from QSettings.
455
456        Returns:
457            List of valid, unique recent-folder paths.
458        """
459        value = self._settings.value(RECENT_DATA_FOLDERS_KEY, [])
460        if value is None:
461            raw: list[str] = []
462        elif isinstance(value, str):
463            raw = [value]
464        else:
465            raw = [str(v) for v in value if v]
466
467        unique: list[str] = []
468        seen: set[str] = set()
469        for folder in raw:
470            try:
471                path_str = str(Path(folder).resolve())
472            except Exception:
473                continue
474            if path_str in seen or not Path(path_str).exists():
475                continue
476            seen.add(path_str)
477            unique.append(path_str)
478        return unique[:MAX_RECENT_DATA_FOLDERS]
479
480    def _save(self) -> None:
481        """Persist the current recent-folder list."""
482        self._settings.setValue(RECENT_DATA_FOLDERS_KEY, self._folders)
483
484    @staticmethod
485    def _format_label(folder_path: str) -> str:
486        """Build the menu label for one recent folder.
487
488        Args:
489            folder_path: Absolute folder path.
490
491        Returns:
492            Formatted menu label.
493        """
494        path = Path(folder_path)
495        parent_name = path.parent.name or "/"
496        return f"{path.name}    [{parent_name}]"
